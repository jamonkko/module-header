#!/usr/bin/env node
'use strict';
const meow = require('meow');
const mh = require('./');

const cli = meow(`
  Usage:
    $ mh <file pattern 1> <file pattern 2> ...

  Example:
    $ mh dist/*.js
`);

if (cli.input.length === 0) {
  console.error('No files provided.');
  process.exit(1);
}

mh('{' + cli.input.join(',') + '}', cli.pkg);
